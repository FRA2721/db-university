/*** QUERY CALLS ***/

QUERY_1

SELECT `students`.`name`, `students`.`surname`, `degrees`.`name` AS `degree`
FROM `students`
INNER JOIN `degrees`
ON `students`.`degree_id` = `degrees`.`id`
WHERE `degrees`.`name` = "Corso di Laurea in Economia";
+------------------------------------------------------------------------------------------+

INNER JOIN `courses`
ON `courses`.`id` = `course_teacher`.`course_id`

`teachers`.`surname` = "Amato";

FROM `students`
